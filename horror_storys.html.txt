  <button type="submit">أرسل القصة</button>

<h2>القصص المُرسلة</h2>
<div id="stories"></div>

<script>
  const form = document.getElementById('storyForm');
  const storiesDiv = document.getElementById('stories');
  const messageDiv = document.getElementById('message');

  const stories = [];

  form.addEventListener('submit', (e) => {
    e.preventDefault();
    const title = document.getElementById('title').value.trim();
    const body = document.getElementById('body').value.trim();

    if (!title || !body) {
      messageDiv.textContent = 'رجاءً اكتب عنوان ونص القصة';
      return;
    }

    const newStory = { title, body };
    stories.unshift(newStory);

    form.reset();
    messageDiv.textContent = 'تم إرسال القصة مؤقتًا (لن تحفظ بعد تحديث الصفحة)';
    renderStories();
  });

  function renderStories() {
    storiesDiv.innerHTML = '';
    if (stories.length === 0) {
      storiesDiv.innerHTML = '<p>لا توجد قصص حتى الآن.</p>';
      return;
    }
    stories.forEach((story) => {
      const article = document.createElement('article');
      const h3 = document.createElement('h3');
      h3.textContent = story.title;
      const p = document.createElement('p');
      p.textContent = story.body;
      article.appendChild(h3);
      article.appendChild(p);
      storiesDiv.appendChild(article);
    });
  }
</script>
